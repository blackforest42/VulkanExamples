# Compile shaders

glslc simple.vert -o simple.vert.spv
glslc blackhole.frag -o blackhole.frag.spv
glslc brightness.frag -o brightness.frag.spv
glslc downsample.frag -o downsample.frag.spv
glslc upsample.frag -o upsample.frag.spv
glslc blend.frag -o blend.frag.spv



# Use `toktx` to create a .ktx cubemap from 6 .png files
toktx --target_type RGBA --cubemap  cubemap.ktx  .\left.png .\right.png .\top.png .\bottom.png .\back.png .\front.png 

## Output
toktx warning! Converting gamma 2.2f to sRGB. Use --assign-oetf srgb to force treating input as sRGB.
