# Use `toktx` to create a .ktx cubemap from 6 .png files
toktx --assign_oetf srgb --target_type RGBA --cubemap  cartoon_sky_cubemap.ktx  .\px.png .\nx.png .\py.png .\ny.png .\pz.png .\nz.png 

toktx --target_type R  --assign_oetf linear iceland_heightmap.ktx .\iceland_heightmap.png
